Control Structures:


if x > 0:
    print("Positive")
elif x == 0:
    print("Zero")
else:
    print("Negative")

for i in range(5):
    print(i)

while True:
    response = input("Enter 'quit' to exit: ")
    if response == 'quit':
        break
    else:
        continue

Functions:


def greet(name):
    return "Hello, " + name

print(greet("Alice"))

def multiply(x, y=2):
    return x * y

print(multiply(3))
print(multiply(3, 4))

Classes and Objects:


class Car:
    def __init__(self, make, model):
        self.make = make
        self.model = model

    def display_info(self):
        print(f"Car: {self.make} {self.model}")

car1 = Car("Toyota", "Camry")
car1.display_info()
